#!/usr/bin/env python
# -*- coding: utf-8 -*-

import argparse
import os
import logging
import yaml
import pytoml as toml
import datetime
import re

default_logging_level = logging.WARNING #日志级别
default_timezome_offset = +datetime.timedelta(hours=8) #默认时区


class Logger(object):
    def __init__(self, name):
        logformat = '%(name)s: [%(levelname)s] %(message)s'

        self.logger = logging.getLogger(name or __name__)
        self.logger.setLevel(default_logging_level)
        myhandler = logging.StreamHandler()
        myhandler.setFormatter(logging.Formatter(logformat))
        self.logger.addHandler(myhandler)


class Hexo2Hugo(object):
    def __init__(self, hexopost, hugopost):
        self.hexo_path = os.path.expanduser(hexopost)
        self.hugo_path = os.path.expanduser(hugopost)
        self.logger = Logger("Hexo2Hugo").logger

        self._find_all_files()

    def go(self):
        self.number = 0
        for post in self._process_files(): #对hexo里面的 yaml front matter 格式转换，对正文中 html 格式转换
            name = post['name'] #hexo文件名
            meta = post['meta'] #元数据，即原来写在yaml front matter里的东西
            body = post['body'] #对html处理后的正文

            with open(os.path.join(self.hugo_path, name), 'w+') as fp:
                self.logger.info("{} 转换完成！\ntoml front matter:\n+++\n{}+++\n".format(name, meta)) #打印日志
                fp.writelines("+++\n{}+++\n\n{}".format(meta, body)) #往 hugo 目录里写文件，元数据前后加上+++，形成toml front matter 格式，空两行写文章正文
                self.number+=1
        self.logger.info("共 {} 篇文章，转换成功 {} 篇，转换失败 {} 篇。".format(len(self.files) ,self.number, len(self.files) - self.number)) #打印日志


    def _find_all_files(self): #拿到hexo下的所有md文件
        self.files = []
        for file in os.listdir(self.hexo_path):
            if os.path.isfile(os.path.join(self.hexo_path, file)):
                self.files.append(file)

    def _extract_date_from_filename(self, filename): #根据文件元数据修改日期格式
        m = re.match('^(\d+-\d+-\d+)-.*$', filename)
        if m:
            return m.group(1)

    def _process_files(self):
        for hexo_file in self.files: #遍历hexo下所有md文件
            date_from_filename = self._extract_date_from_filename(hexo_file)
            with open(os.path.join(self.hexo_path, hexo_file), 'r') as fp: #拼接文件绝对路径，打开文件
                meta_yaml = '' #yaml front matter
                body = '' #自己写的文章正文
                
                is_meta = True #临界区设置
                is_first_line = True #临界区设置
                is_in_pre = False #临界区设置
                
                for line in fp: #按行遍历文件
                    if is_first_line:
                        is_first_line = False
                        if line == '---\n': #判断是不是yaml front matter
                            continue

                    if is_meta:
                        if line == '---\n': #判断是不是yaml front matter
                            is_meta = False
                        else:
                            meta_yaml += line #如果不是最后一行，把数据写到meta_yaml里
                    else: #前面是对front matter处理，后面是对正文处理
                        if re.search(r'.html$', hexo_file): #对正文出现的的html格式处理
                            if re.search(r'^<pre', line):
                                is_in_pre = True

                            if re.search(r'^</pre', line):
                                is_in_pre = False

                            if not is_in_pre and not re.match(r'^$', line) and not re.search(r'<br */>$', line):
                                body += line[:-1] + "<br />\n"
                            else:
                                body += line
                        else:
                            body += line
                #利用python库函数进行格式转换, {'title': '驱动开发', 'date': '2020-07-21 11:42', 'categories': '操作系统', 'tags': '驱动开发', 'typora-root-url': '../'}
                meta = yaml.load(meta_yaml)

                self.logger.info("{} 正在转换...".format(hexo_file)) #控制台打印日志

                if 'date' in meta: #日期格式化
                    hexo_date = meta['date']
                    if type(hexo_date) == str:
                        hexo_date = datetime.datetime.strptime(hexo_date, '%Y-%m-%d %H:%M')

                    meta['date'] = hexo_date.replace(tzinfo=datetime.timezone(default_timezome_offset)).isoformat('T') #换成hugo支持的格式
                else:
                    meta['date'] = date_from_filename #如果front matter里面没有指明日期，根据文件创建日期格式化

                meta['description'] = ''
                if 'permalink' in meta: #文章永久链接，在hexo里叫permalink，在hugo里叫slug
                    meta['slug'] = meta['permalink']
                    del meta['permalink']

                if 'layout' in meta: #指定的路径，hexo默认是_posts，换到hugo之后路径都不一样，没用就删掉
                    del meta['layout']

                if 'typora-root-url' in meta: #使用typora写的时候为了图片显示正常，换成hugo，typora不支持toml格式，这里也就没用了，就删掉了
                    del meta['typora-root-url']

                #tab标签解析，这里是按逗号分割的，如果不是这种格式，可以参考下面网址自己修改，感谢作者
                #https://github.com/oouxx/hexo2hugo/blob/master/hexo2hugo.py
                if 'tags' in meta and type(meta['tags']) == str:
                    meta['tags'] = meta['tags'].split(',')

                meta_toml = toml.dumps(meta) #把 front matter 转成 toml格式
                yield({'name': hexo_file, 'meta': meta_toml, 'body': body})

def main(args):
    hexo = Hexo2Hugo(args.srcpost, args.descpost)
    hexo.go()


if __name__ == '__main__':
    parser = argparse.ArgumentParser()
    parser.add_argument('--srcpost', help='Hexo posts directory')
    parser.add_argument('--descpost', help='Hugo posts directory')
    parser.add_argument('--verbose' , help='Output level, default debug', action='store_true')

    args = parser.parse_args()

    if args.verbose:
        default_logging_level = logging.DEBUG

    main(args)

图片迁移

  • 同步脚本
#!/bin/bash
#1. 判断参数个数
if [ $# -lt 2 ]
then
    echo Not Enough Arguement!
    exit;
fi

#2. 遍历所有目录，挨个发送
for file in $1
do
    #4.5 判断文件是否存在
    if [ -e $file ]
    then
        #3. 获取父目录
        pdir=$(cd -P $(dirname $file); pwd)
        #4. 获取当前文件的名称
        fname=$(basename $file)
        #5. 拷贝  
        echo =========    正在同步    =========
        mkdir -p $2
        rsync -avh $pdir/$fname $2       
    else
        echo $file does not exists!
    fi
done
