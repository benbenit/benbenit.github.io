编译

ben@ben:~$ cd src
ben@ben:~/src$ sudo apt install -y openjdk-8-jdk

ben@ben:~/src$ java -version

ben@ben:~/src$ sudo apt install -y gcc g++ gdb make ccache libfreetype6-dev libcups2-dev libx11-dev libxext-dev libxrender-dev libxrandr-dev libxtst-dev libxt-dev libasound2-dev libffi-dev autoconf libfontconfig1-dev

#安装mercurial是为了使用hg
ben@ben:~/src$ sudo apt install mercurial
ben@ben:~/src$ hg clone http://hg.openjdk.java.net/jdk8u/jdk8u-dev
ben@ben:~/src$ cd jdk8u-dev/

ben@ben:~/src/jdk8u-dev$ chmod +x ./get_source.sh 
ben@ben:~/src/jdk8u-dev$ chmod +x ./configure 
ben@ben:~/src/jdk8u-dev$ ./get_source.sh

#debug级别，是否压缩，jvm级别，机器位数，用几核编译，起几个job
ben@ben:~/src/jdk8u-dev$ ./configure --with-debug-level=slowdebug --disable-zip-debug-info -with-jvm-variants=server --with-target-bits=64 --with-num-cores=4 --with-jobs=4

ben@ben:~/src/jdk8u-dev$ vim ./hotspot/make/linux/makefiles/gcc.make
//204行，注掉
# WARNINGS_ARE_ERRORS = -Werror

# 下载Bear工具,为了方便使用clion调试

# 构建，并使用bear工具生成Compilation Database
ben@ben:~/src/jdk8u-dev$ bear make all

ben@ben:~/src/jdk8u-dev$ ls ./build/linux-x86_64-normal-server-slowdebug/images/j2sdk-image/
ASSEMBLY_EXCEPTION  bin  demo  include  jre  lib  LICENSE  man  release  sample  src.zip  THIRD_PARTY_README

//建软链
ben@ben:~/src/jdk8u-dev$ ln -s /home/ben/src/jdk8u-dev/build/linux-x86_64-normal-server-slowdebug/images/j2sdk-image /home/ben/apps/java8

//JAVA_HOME
ben@ben:~/src/jdk8u-dev$ vim ~/.bashrc

export PATH=$PATH:$JAVA_HOME/bin

Hadoop源码编译

环境

环境 版本
Ubuntu 18.04.4 LTS amd64
JDK 1.8.0_231
Maven 3.6.3
Protobuf 2.5.0
Snappy 1.1.3
lzo 2.08
lz4 0.0~r131-2ub
zlib1g (zlib) 1.2.11
  • root用户，安装protobuf
root@ben:~# cd /usr/local/

root@ben:/usr/local# tar -zxf /home/ben/src/protobuf-2.5.0.tar.gz -C /usr/local/src/
root@ben:/usr/local# cd /usr/local/src/protobuf-2.5.0/
root@ben:/usr/local/src/protobuf-2.5.0# ./configure --prefix /usr/local/protobuf
root@ben:/usr/local/src/protobuf-2.5.0# make && make install
  • root用户，安装snappy
root@ben:/usr/local/bin# tar -zxf /home/ben/src/snappy-1.1.3.tar.gz -C /usr/local/bin
root@ben:/usr/local/bin# chown -R root:root /usr/local/bin/snappy-1.1.3
root@ben:/usr/local/bin# cd /usr/local/bin/snappy-1.1.3
root@ben:/usr/local/bin/snappy-1.1.3# ./configurate
root@ben:/usr/local/bin/snappy-1.1.3# make && make install
  • 用户ben
#先安装ruby，再安装zlib
sudo apt install -y ruby
sudo apt install -y zlib1g zlib1g-devel

#安装openssl1.0，使用apt默认安装的是2.0，会报错，先卸载2.0
sudo apt remove libssl-dev
sudo apt install -y libssl1.0-dev

#安装其他
sudo apt install g++ cmake liblzo2-dev liblz4-dev bzip2 libbz2-dev
